<p style="text-align: center;">JSP 기본개념/요소정리 + 예제</p>
<hr contenteditable="false" />
<p style="text-align: left;">1. JSP : (Java Server Pages)는 HTML 문서 안에 Java 코드를 직접 작성할 수 있는 기술이자 페이지</p>
<p style="text-align: left;">- Servlet에서 응답 화면(HTML)을 만드는 복잡한 과정을 단순화해주며, 비즈니스 로직은 Servlet, 화면은 JSP가 담당하는 MVC 패턴의 기초 역할을 한다<br />- JSP는 내부적으로 JSP -&gt; Servlet으로 변환되어 실행되며 request, response, session 같은 내장 객체를 기본적으로 제공</p>
<p style="text-align: left;">&nbsp;</p>
<p style="text-align: left;">2. JSP 요소</p>
<p style="text-align: left;">(1) 스크립트 : JSP&nbsp;페이지에서&nbsp;자바&nbsp;코드를&nbsp;기술할&nbsp;수&nbsp;있는&nbsp;방법 <br />-&nbsp;스크립틀릿(Scriptlet)&nbsp;:&nbsp;&lt;%&nbsp;자바코드&nbsp;%&gt;&nbsp;:&nbsp;JSP에서&nbsp;자바&nbsp;코드를&nbsp;직접&nbsp;작성하는&nbsp;가장&nbsp;기본적인&nbsp;방식입니다. <br />- 표현식(Expression) : &lt;%= 변수 %&gt; : 자바 변수에 저장된 값을 화면에 출력할 때 사용합니다.<br /><br /></p>
<pre class="bash" id="code_1759129653086"><code>&lt;%
    int sum = 0;
    for(int i=1; i&lt;=100; i++){
        sum += i;
    }
%&gt;

&lt;p&gt;
    스크립틀릿 출력: &lt;% out.println(sum); %&gt; &lt;br&gt;
    표현식 출력: &lt;%= sum %&gt;
&lt;/p&gt;

&lt;%
    String[] nameArr = {"지원", "수민", "지수"};
%&gt;

&lt;h5&gt;배열 길이 : &lt;%=nameArr.length %&gt;&lt;/h5&gt;
&lt;h5&gt;배열 값 : &lt;%=String.join(", ", nameArr) %&gt;&lt;/h5&gt;

&lt;ul&gt;
    &lt;% for(int i=0; i&lt;nameArr.length; i++){ %&gt;
        &lt;li&gt;&lt;%= nameArr[i] %&gt;&lt;/li&gt;
    &lt;% } %&gt;
&lt;/ul&gt;</code></pre>
<p style="text-align: left;">-&gt; 반복문을&nbsp;이용해&nbsp;HTML&nbsp;요소를&nbsp;동적으로&nbsp;생성하는&nbsp;예제</p>
<p style="text-align: left;">&nbsp;</p>
<p style="text-align: left;">(2) 지시어 : JSP&nbsp;페이지&nbsp;전체에&nbsp;영향을&nbsp;주는&nbsp;속성을&nbsp;선언할&nbsp;때&nbsp;사용한다 <br /><br />형식 : &lt;%@ page | include | taglib 속성=값 ... %&gt;</p>

import java.io.IOException;

import jakarta.servlet.RequestDispatcher;
import jakarta.servlet.ServletException;
import jakarta.servlet.annotation.WebServlet;
import jakarta.servlet.http.HttpServlet;
import jakarta.servlet.http.HttpServletRequest;
import jakarta.servlet.http.HttpServletResponse;

/**
 * 네트워크열어서 http요청 받아주는 객체
 * HttpServletRequest request = new HttpServletRequest();
 * //request객체에 네트워크로부터 전달된 데이터를 전부 파싱해서 넣기
 * 
 * PizzaOrderServlet pizzaorder = new PizzaOrderServlet();
 * pizzaorder.doPost(request);
 */
@WebServlet("/confirmPizza.do")
public class PizzaOrderServlet extends HttpServlet {
	private static final long serialVersionUID = 1L;
       
    /**
     * @see HttpServlet#HttpServlet()
     */
    public PizzaOrderServlet() {
        super();
        // TODO Auto-generated constructor stub
    }

	/**
	 * @see HttpServlet#doGet(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doGet(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		//피자주문 요청 처리 -&gt; 결제페이지
		
		//요청시 전달한 데이터를 추출, 데이터를 가공 -&gt; 변수나 객체 넘겨받은 값을 기록
		String name = request.getParameter("userName");
		String phone = request.getParameter("phone");
		String address = request.getParameter("address");
		String message = request.getParameter("message");
		
		String pizza = request.getParameter("pizza");
		String[] toppingArr = request.getParameterValues("topping"); //메서드명이 동일한데 당연히 같은 형식의 값을 리턴할 수 없죠.
		String[] sideArr = request.getParameterValues("side");
		String payment = request.getParameter("payment");
		
		//service &gt; dao &gt; db
		
		int price = 0;
		switch(pizza) {
			case "콤비네이션": price += 20000; break;
			case "포테이토":
			case "고구마" : price += 25000; break;
			default : price += 30000; break;
		}
		
		if(toppingArr != null) {
			for(String topping : toppingArr) {
				switch(topping) {
				case "베이컨":
				case "파인애플" : price += 3000; break;
				case "치즈크러스트":
				case "치즈바이트": price += 2000; break;
				default: price += 1000;
				}
			}
		}
		
		if(sideArr != null) {
			for(String side : sideArr) {
				switch(side) {
				case "콜라":
				case "사이다" : price += 3000; break;
				case "핫소스":
				case "파마산": price += 2000; break;
				default: price += 1000;
				}
			}
		}
		
		//응답페이지에 필요한 정보를 request의 attribute영역에 담아서 전달
		request.setAttribute("name", name);
		request.setAttribute("phone", phone);
		request.setAttribute("address", address);
		request.setAttribute("message", message);
		
		request.setAttribute("pizza", pizza);
		request.setAttribute("toppingArr", toppingArr);
		request.setAttribute("sideArr", sideArr);
		request.setAttribute("payment", payment);
		request.setAttribute("price", price);
		
		//요청처리 후 사용자가 보게될 응답페이지를 만들어서 전달.
		request.getRequestDispatcher("views/pizza/pizzaPayment.jsp").forward(request, response);;
	}

	/**
	 * @see HttpServlet#doPost(HttpServletRequest request, HttpServletResponse response)
	 */
	protected void doPost(HttpServletRequest request, HttpServletResponse response) throws ServletException, IOException {
		doGet(request, response);
	}

}</code></pre>

<p>배운점 &amp; 느낀점</p>
<p>JSP를&nbsp;배우면서&nbsp;가장&nbsp;크게&nbsp;느낀&nbsp;점은&nbsp;단순히&nbsp;HTML&nbsp;안에&nbsp;자바&nbsp;코드를&nbsp;섞어서&nbsp;쓰는&nbsp;기술이&nbsp;아니라&nbsp;서블릿과&nbsp;역할을&nbsp;분리해&nbsp;MVC&nbsp;구조를&nbsp;이해할&nbsp;수&nbsp;있게&nbsp;해주는&nbsp;기반이라는&nbsp;것이었다.&nbsp;스크립틀릿과&nbsp;표현식을&nbsp;통해&nbsp;화면에서&nbsp;데이터를&nbsp;다루는&nbsp;방식을&nbsp;경험하면서&nbsp;코드&nbsp;실행과&nbsp;출력이&nbsp;어떻게&nbsp;다른지&nbsp;자연스럽게&nbsp;익힐&nbsp;수&nbsp;있었고&nbsp;지시어를&nbsp;활용해&nbsp;페이지의&nbsp;속성을&nbsp;지정하거나&nbsp;다른&nbsp;JSP를&nbsp;포함시켜&nbsp;쓰면서&nbsp;코드&nbsp;재사용과&nbsp;유지보수의&nbsp;편리함도&nbsp;알게&nbsp;되었다.&nbsp;특히&nbsp;피자&nbsp;주문&nbsp;프로젝트를&nbsp;해보면서&nbsp;사용자가&nbsp;폼에&nbsp;입력한&nbsp;값이&nbsp;서블릿에서&nbsp;처리되고&nbsp;다시&nbsp;JSP&nbsp;화면으로&nbsp;결과가&nbsp;전달되는&nbsp;흐름을&nbsp;직접&nbsp;체험할&nbsp;수&nbsp;있었는데,&nbsp;이를&nbsp;통해&nbsp;단순한&nbsp;문법&nbsp;학습을&nbsp;넘어&nbsp;실제&nbsp;웹&nbsp;애플리케이션이&nbsp;어떻게&nbsp;동작하는지를&nbsp;더&nbsp;깊이&nbsp;이해할&nbsp;수&nbsp;있었다.</p>
