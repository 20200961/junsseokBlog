<p>(1) BankAccount 시스템&nbsp;</p>
<p>BankAccount.java</p>
<pre class="bash" id="code_1754984683306"><code>package com.kh.object.ex2;

import java.lang.annotation.Target;

public class BankAccount {
    int balance = 0;

    public BankAccount() {
        System.out.println("생성자 실행됨.");
    }

    public BankAccount(int balance) {
        this.balance = balance;
    }

    void deposit(int amount) {
        balance += amount;
    }

    void withdraw(int amount) {
        balance -= amount;
    }

    void checkMyBalance() {
        System.out.println("잔액 :" + balance);
    }

    void transfer(BankAccount target, int amount) {
        balance += amount;
        target.balance -= 1000;
    }
}</code></pre>
<p>&nbsp;</p>
<p>Run.java</p>
<pre class="bash" id="code_1754984729642"><code>package com.kh.object.ex2;

public class Run {

   public static void main(String[] args) {
      BankAccount ac1 = new BankAccount();
      BankAccount ac2 = ac1;
      
      ac1.deposit(5000);
      ac2.withdraw(1000);
      ac1.checkMyBalance();
      
      
      ac2 = new BankAccount(2000);

      
      ac1.transfer(ac2, 1000);
      
      
   }

}</code></pre>

<p>(2) Math -&gt; 수학 문제&nbsp;</p>
<p><span style="color: #333333; text-align: start;">Math.java</span></p>
<pre class="bash" id="code_1754984749545"><code>package com.kh.object.ex3;

public class Math {

	int adder(int n1) {
		return n1 + n1;
	}
		
	int adder(int n1,int n2) {
		return n1+n2;
	}
		
	double adder(double n1,double n2) {
		return n1 + n2;
	}
	
	String adder(int n1, String ch) {
		String addResult = n1 + ch;
		return addResult;
	}
	
	String adder(String ch, int n1) {
		String addResult = ch + n1;
		return addResult;
	}


}</code></pre>
<p>&nbsp;</p>
<p>Run.java</p>
<pre class="bash" id="code_1754984792034"><code>package com.kh.object.ex3;

public class Run {

	public static void main(String[] args) {
		Math m = new Math();
		System.out.println(m.adder(10,20));
		System.out.println(m.adder(10));
		System.out.println(m.adder(10.0,30.0));
		System.out.println(m.adder('a','b'));
		System.out.println(m.adder(10,"살입니다."));
